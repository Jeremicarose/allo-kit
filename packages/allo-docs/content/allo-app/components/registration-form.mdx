# RegistrationForm

The RegistrationForm component can be used to register projects or applications.

```tsx
import { RegistrationForm } from "~/components/registration/registration-form";

export default function CreateProjectPage() {
  const { YourStrategy } = useContracts();
  const account = useAccount();
  const router = useRouter();
  const upload = useIpfsUpload();
  const register = useRegister({ strategyAddress: YourStrategy?.address });
  return (
    <RegistrationForm
      defaultValues={{}}
      onSubmit={async (values) => {
        const { cid: metadataURI } = await upload.mutateAsync(values.metadata);
        const { project } = await register.mutateAsync([account.address, metadataURI, ""]);
        return router.push(`/projects/${project});
      }}
    />
  );
}
```
